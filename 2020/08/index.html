func isValidBST(root *TreeNode) bool {

func helper(root *TreeNode, min, max int) bool {
	if root == nil {
		return true
	}
	if min != -1 && root.Val <= min {
		return false
	}
	if max != -1 && root.Val >= max {
		return false
	}
	if !helper(root.Left, min, root.Val) {
		return false
	}

	if !helper(root.Right, root.Val, max) {
		return false
	}
	return true
}
