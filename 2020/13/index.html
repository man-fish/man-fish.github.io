func dfs(m, n int, dp [][]int) int {

		return 1

func dfs(grid, dp [][]int, x, y int) int {
	m := len(grid[0]) - 1
	n := len(grid) - 1
	if y == n && x == m {
		return grid[y][x]
	}
	if dp[y][x] != 0 {
		return dp[y][x]
	}

}

	if x == 0 || y == 0 {
		return 0
	}

func maximalRectangle(matrix [][]byte) int {
	n := len(matrix)
	if n <= 0 {
		return 0
	}
	m := len(matrix[0])
	dp := make([][]int, n)
	for idx := range dp {
		dp[idx] = make([]int, m)
	}
	maxS := 0
	for i := 0; i < n; i++ {
		for j := 0; j < m; j++ {
			if matrix[i][j] != '1' {
				dp[i][j] = 0
				continue
			}
			if j == 0 {
				dp[i][j] = 1
			} else {
				dp[i][j] = dp[i][j-1] + 1
			}
			minWidth := dp[i][j]
			for k := i; k >= 0; k-- {
				minWidth = min(dp[k][j], minWidth)
				maxS = max(maxS, minWidth*(i-k+1))
			}
		}
	}
	return maxS
}

func helper(root *TreeNode, min, max int) bool {
    if root == nil {
        return true
    }
    if min != -1 && root.Val <= min {
        return false
    }
    if max != -1 && root.Val >= max {
        return false 
    }

        return false
    }

        return false
    }
    return true
}
